def is_string_dtype(arr_or_dtype) -> bool:
    """
    Check whether the provided array or dtype is of the string dtype.


    Parameters
    ----------
    arr_or_dtype : array-like
        The array or dtype to check.


    Returns
    -------
    boolean
        Whether or not the array or dtype is of the string dtype.


    Examples
    --------
    >>> is_string_dtype(str)
    True
    >>> is_string_dtype(object)
    True
    >>> is_string_dtype(int)
    False
    >>>
    >>> is_string_dtype(np.array(['a', 'b']))
    True
    >>> is_string_dtype(pd.Series([1, 2]))
    False
    """
    # TODO: gh-15585: consider making the checks stricter.
    def condition(dtype) -> bool:
        return dtype.kind in ("O", "S", "U") and not is_excluded_dtype(dtype)


    def is_excluded_dtype(dtype) -> bool:
        """
        These have kind = "O" but aren't string dtypes so need to be explicitly excluded
        """
        is_excluded_checks = (is_period_dtype, is_interval_dtype)
        return any(is_excluded(dtype) for is_excluded in is_excluded_checks)


    return _is_dtype(arr_or_dtype, condition)

class _Numeric:
    """Class to easily allow comparing numbers"""

    def __init__(self, specifier):
        self.specifier = int(specifier)

    def __repr__(self):
        return repr(self.specifier)

    def __eq__(self, other):
        if isinstance(other, _Numeric):
            return self.specifier == other.specifier
        elif isinstance(other, int):
            return self.specifier == other

        return False

    def __ne__(self, other):
        return not self.__eq__(other)

    def __lt__(self, other):
        if isinstance(other, _Numeric):
            return self.specifier < other.specifier
        elif isinstance(other, int):
            return self.specifier < other
        elif isinstance(other, _Alpha):
            return True

        raise ValueError

    def test_numeric_greater_than_itself_is_false(self):
        """
        Ensures that a _Numeric instance is not considered greater than itself.
        This directly targets the bug in _Numeric.__gt__ where it was incorrectly
        implemented as 'not self.__lt__(other)', causing '5 > 5' to return True.
        """

        # In the buggy version:
        # num.__lt__(num) -> False (since 5 < 5 is False)
        # num.__gt__(num) -> not False -> True
        # So, 'num > num' evaluates to True.
        # The assertion 'assert not (num > num)' will then fail because 'not True' is False.
        # In the fixed version, num.__gt__(num) returns False, so 'not False' is True, and the test passes.

def check_required_arguments(argument_spec, module_parameters):
    """Check all paramaters in argument_spec and return a list of parameters
    that are required but not present in module_parameters
    """
    missing = []
    if argument_spec is None:
        return missing

    for (k, v) in argument_spec.items():
        required = v.get('required', False)
        if required and k not in module_parameters:
            missing.append(k)

    if missing:
        msg = "missing required arguments: %s" % ", ".join(missing)
        raise TypeError(msg)

    return missing

        return False
    try:
        return gpr is obj[gpr.name]
    except (KeyError, IndexError):
        return False

def update_or_append_format(formats, formats_dict, representation_id, f):
    try:
        existing_format = next(
            fo for fo in formats
            if fo['format_id'] == representation_id
        )
    except StopIteration:
        full_info = formats_dict.get(representation_id, {}).copy()
        full_info.update(f)
        formats.append(full_info)
    else:
        existing_format.update(f)

    return formats

@pytest.mark.parametrize(
    "test_name, tokens, expected_output",
    [
        (
            "stops at the first OP ']' in a sequence",
            [
                _make_token(tokenize.NAME, 'column_name'),
                _make_token(tokenize.OP, '='),
                _make_token(tokenize.NUMBER, '100'),
                _make_token(tokenize.OP, ']'),  # Termination token
                _make_token(tokenize.NAME, 'ignored_part')
            ],
            'column_name=100'
        ),
        (
            "returns an empty string if OP ']' is the very first token",
            [
                _make_token(tokenize.OP, ']'),  # Termination token
                _make_token(tokenize.NAME, 'some_name')
            ],
            ''
        ),
        (
            "stops correctly when OP ']' is the last token",
            [
                _make_token(tokenize.NAME, 'filter_field'),
                _make_token(tokenize.OP, '>'),
                _make_token(tokenize.STRING, "'value'"),
                _make_token(tokenize.OP, ']')  # Termination token
            ],
            "filter_field>'value'"
        ),
        (
            "returns None if no OP ']' token is found",
            [
                _make_token(tokenize.NAME, 'name'),
                _make_token(tokenize.OP, '.'),
                _make_token(tokenize.NAME, 'startswith'),
                _make_token(tokenize.OP, '('),
                _make_token(tokenize.STRING, "'prefix'"),
                _make_token(tokenize.OP, ')')
            ],
            None
        ),
        (
            "returns None for an empty list of tokens",
            [],
            None
        ),
        (
            "includes non-OP ']' strings and stops at actual OP ']'",
            [
                _make_token(tokenize.NAME, 'item'),
                _make_token(tokenize.STRING, ']'),  # This ']' is part of a string, not an operator
                _make_token(tokenize.OP, '='),
                _make_token(tokenize.NUMBER, '1'),
                _make_token(tokenize.OP, ']'),  # This ']' is an operator, should stop here
                _make_token(tokenize.NAME, 'should_be_ignored')
            ],
            'item]=1'
        ),
        (
            "returns None if only non-OP ']' strings are present",
            [
                _make_token(tokenize.NAME, 'my_list'),
                _make_token(tokenize.STRING, ']'),
                _make_token(tokenize.NUMBER, '0')
            ],
            None
        ),
        (
            "handles complex filter expressions",
            [
                _make_token(tokenize.NAME, 'data'),
                _make_token(tokenize.OP, '.'),
                _make_token(tokenize.NAME, 'loc'),
                _make_token(tokenize.OP, '['),
                _make_token(tokenize.NAME, 'data'),
                _make_token(tokenize.OP, '.'),
                _make_token(tokenize.NAME, 'column'),
                _make_token(tokenize.OP, '=='),
                _make_token(tokenize.NUMBER, '5'),
                _make_token(tokenize.OP, ']') # Termination
            ],
            'data.loc[data.column==5'
        ),
    ]
)
def test_parse_filter_functionality(test_name, tokens, expected_output):
    """
    Test the parse_filter function for various scenarios including
    correct termination, edge cases like empty input or first/last token being ']',
    and handling of non-OP ']' characters.
    """
    assert parse_filter(tokens) == expected_output

def strip_jsonp(code):
    return re.sub(
        r'(?s)^[a-zA-Z0-9_]+\s*\(\s*(.*)\);?\s*?(?://[^\n]*)*$', r'\1', code)

def test_strip_jsonp_basic():
    # Case 1: standard JSONP
    assert strip_jsonp("callback({\"a\":1});") == "{\"a\":1}"

    # Case 2: with whitespace
    assert strip_jsonp("cb ( {\"b\":2} ) ;") == "{\"b\":2}"

    # Case 3: with trailing comment
    assert strip_jsonp("cb({\"c\":3}); // comment") == "{\"c\":3}"

    # Case 4: no wrapping → unchanged
    assert strip_jsonp("{\"d\":4}") == "{\"d\":4}"
