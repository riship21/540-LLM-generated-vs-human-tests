            np.dtype(dtype)

        assert not np.str_ == dtype

    def test_pickle(self, dtype):
        # make sure our cache is NOT pickled

        # clear the cache

        result = tm.round_trip_pickle(dtype)
        assert not len(dtype._cache)
        assert result == dtype
