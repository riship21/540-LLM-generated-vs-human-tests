class TestNormalizeUsername(unittest.TestCase):


    def test_basic(self):
        self.assertEqual(normalize_username("Strongest Soldier"), "strongest-soldier")


    def test_leading_trailing_whitespace(self):
        self.assertEqual(normalize_username("  harry  "), "harry")


    def test_multiple_internal_spaces(self):
        self.assertEqual(normalize_username("john   doe"), "john-doe")


    def test_already_lowercase(self):
        self.assertEqual(normalize_username("bob"), "bob")


    def test_uppercase(self):
        self.assertEqual(normalize_username("TAY"), "tay")


    def test_empty_string(self):
        self.assertEqual(normalize_username(""), "")


    def test_only_spaces(self):
        self.assertEqual(normalize_username("   "), "")


    def test_three_words(self):
        self.assertEqual(normalize_username("Tyrone The Realest"), "tyrone-the-realest")


    def test_mixed_case_with_spaces(self):
        self.assertEqual(normalize_username("  John DOE  "), "john-doe")




class TestMergeIntervals(unittest.TestCase):


    def test_empty_list(self):
        self.assertEqual(merge_intervals([]), [])


    def test_single_interval(self):
        self.assertEqual(merge_intervals([(1, 3)]), [(1, 3)])


    def test_no_overlap(self):
        self.assertEqual(merge_intervals([(1, 2), (4, 6)]), [(1, 2), (4, 6)])


    def test_overlapping(self):
        self.assertEqual(merge_intervals([(1, 3), (2, 5)]), [(1, 5)])


    def test_touching_intervals(self):
        self.assertEqual(merge_intervals([(1, 3), (3, 5)]), [(1, 5)])


    def test_one_fully_contained(self):
        self.assertEqual(merge_intervals([(1, 10), (2, 5)]), [(1, 10)])


    def test_unsorted_input(self):
        self.assertEqual(merge_intervals([(4, 6), (1, 3)]), [(1, 3), (4, 6)])


    def test_multiple_merges(self):
        result = merge_intervals([(1, 4), (2, 5), (7, 9), (8, 10)])
        self.assertEqual(result, [(1, 5), (7, 10)])


    def test_all_overlap_into_one(self):
        self.assertEqual(merge_intervals([(1, 5), (2, 6), (3, 7)]), [(1, 7)])


    def test_identical_intervals(self):
        self.assertEqual(merge_intervals([(2, 4), (2, 4)]), [(2, 4)])
