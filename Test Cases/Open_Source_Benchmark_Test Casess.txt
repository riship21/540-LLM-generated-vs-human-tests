def canonicalize_name(name: str, *, validate: bool = False) -> NormalizedName:
    if validate and not _validate_regex.fullmatch(name):
        raise InvalidName(f"name is invalid: {name!r}")


    value = name.lower().replace("_", "-").replace(".", "-")


    while "--" in value:
        value = value.replace("--", "-")


    return cast("NormalizedName", value)

# --- Minimal implementations for dependencies to make the function runnable ---
# These simulate the expected behavior of Version objects and the _TrimmedRelease helper.


class InvalidVersion(ValueError):
    """Custom exception for invalid versions, similar to packaging.version.InvalidVersion."""
    pass


class Version:
    """A simplified Version class for testing purposes."""
    def __init__(self, version_str: str):
        if not isinstance(version_str, str):
            raise TypeError("Version string must be a string.")
        self._original_str = version_str
        try:
            # Simulate parsing into numeric parts, raising InvalidVersion for non-numeric components
            self.release = tuple(int(p) for p in version_str.split('.'))
        except ValueError as e:
            raise InvalidVersion(f"Invalid version string: '{version_str}'") from e


    def __str__(self) -> str:
        return self._original_str


    def __repr__(self) -> str:
        return f"Version('{self._original_str}')"


    def __eq__(self, other) -> bool:
        if isinstance(other, Version):
            return self._original_str == other._original_str
        return NotImplemented


class _TrimmedRelease:
    """A simplified helper to simulate stripping trailing .0 components."""
    def __init__(self, version_obj: Version):
        release_parts = list(version_obj.release)
        # Strip all trailing .0 components, but ensure at least one component remains (e.g., '0' for '0.0.0')
        while len(release_parts) > 1 and release_parts[-1] == 0:
            release_parts.pop()
        self._trimmed_str = ".".join(str(p) for p in release_parts)


    def __str__(self) -> str:
        return self._trimmed_str


# --- The function under test (copied for self-containment) ---
def canonicalize_version(
    version: Version | str, *, strip_trailing_zero: bool = True
) -> str:
    if isinstance(version, str):
        try:
            version = Version(version)
        except InvalidVersion:
            return str(version)
    return str(_TrimmedRelease(version) if strip_trailing_zero else version)

def parse_sdist_filename(filename: str) -> tuple[NormalizedName, Version]:
    if filename.endswith(".tar.gz"):
        file_stem = filename[: -len(".tar.gz")]
    elif filename.endswith(".zip"):
        file_stem = filename[: -len(".zip")]
    else:
        raise InvalidSdistFilename(
            f"Invalid sdist filename (extension must be '.tar.gz' or '.zip'):"
            f" {filename!r}"
        )


    name_part, sep, version_part = file_stem.rpartition("-")
    if not sep:
        raise InvalidSdistFilename(f"Invalid sdist filename: {filename!r}")


    name = canonicalize_name(name_part)


    try:
        version = Version(version_part)
    except InvalidVersion as e:
        raise InvalidSdistFilename(
            f"Invalid sdist filename (invalid version): {filename!r}"
        ) from e


    return (name, version)

def smart_truncate(
    string: str,
    max_length: int = 0,
    word_boundary: bool = False,
    separator: str = " ",
    save_order: bool = False,
) -> str:


    string = string.strip(separator)


    if not max_length:
        return string


    if len(string) < max_length:
        return string


    if not word_boundary:
        return string[:max_length].strip(separator)


    if separator not in string:
        return string[:max_length]


    truncated = ''
    for word in string.split(separator):
        if word:
            next_len = len(truncated) + len(word)
            if next_len < max_length:
                truncated += '{}{}'.format(word, separator)
            elif next_len == max_length:
                truncated += '{}'.format(word)
                break
            else:
                if save_order:
                    break
    if not truncated:
        truncated = string[:max_length]
    return truncated.strip(separator)

def test_smart_truncate_word_boundary_and_save_order_interaction():
    """
    Tests the `save_order` parameter's behavior when `word_boundary` is true
    and a long word would cause the total length to exceed `max_length`.
    - If `save_order=True`, truncation should stop before adding the exceeding word.
    - If `save_order=False`, the exceeding word should be skipped, and subsequent
      words should be considered if they fit within `max_length`.
    """
    text = "alpha bravo charlie_very_long delta"
    max_len = 16
    separator = " "


    # Scenario:
    # 1. "alpha" (5 chars) + " " = "alpha " (6 chars)
    # 2. "bravo" (5 chars) + " " = "alpha bravo " (12 chars)
    # 3. "charlie_very_long" (17 chars)
    #    next_len = 12 + 17 = 29. This exceeds max_len (16).


    # With save_order=True: The function should stop at "bravo" because adding
    # "charlie_very_long" would exceed max_len.
    expected_true = "alpha bravo"
    assert smart_truncate(text, max_len, word_boundary=True, separator=separator, save_order=True) == expected_true


    # With save_order=False: The function should skip "charlie_very_long" and
    # continue to the next word ("delta").
    # "alpha bravo " (12 chars) + "delta" (5 chars) = 17. Still exceeds.
    # Uh, wait, my mental trace above for save_order=False: "first third" had max_len=12, first=5, third=5, sep=" ".
    # "first " (6) + "third" (5) = 11. It fit.
    # In this case: "alpha bravo " (12) + "delta" (5) = 17. It would exceed max_len (16).
    # So "delta" should also be skipped.
    # The result should still be "alpha bravo". Let's re-verify my understanding for this case.


    # Re-tracing for save_order=False, max_len=16:
    # 1. 'alpha': truncated = 'alpha ' (len 6)
    # 2. 'bravo': truncated = 'alpha bravo ' (len 12)
    # 3. 'charlie_very_long': next_len = 12 + 17 = 29. 29 > 16. save_order=False, so skip 'charlie_very_long'.
    # 4. 'delta': next_len = 12 + 5 = 17. 17 > 16. save_order=False, so skip 'delta'.
    # Loop finishes.
    # `truncated` is 'alpha bravo '.
    # returns 'alpha bravo'.


    # This means for _this specific_ `max_len`, both `save_order=True` and `save_order=False`
    # would yield the same result. This is not ideal for demonstrating the difference.
    # I need to adjust `max_len` or `text` such that 'delta' *can* fit if 'charlie_very_long' is skipped.


    # Let's adjust `max_len` to 17.
    # text = "alpha bravo charlie_very_long delta"
    # max_len = 17
    # separator = " "


    # Save_order=True:
    # 1. 'alpha': truncated = 'alpha ' (len 6)
    # 2. 'bravo': truncated = 'alpha bravo ' (len 12)
    # 3. 'charlie_very_long': next_len = 12 + 17 = 29. 29 > 17. Break.
    # Returns "alpha bravo"


    # Save_order=False:
    # 1. 'alpha': truncated = 'alpha ' (len 6)
    # 2. 'bravo': truncated = 'alpha bravo ' (len 12)
    # 3. 'charlie_very_long': next_len = 12 + 17 = 29. 29 > 17. Skip.
    # 4. 'delta': next_len = 12 + 5 = 17. 17 == 17.
    #    truncated = 'alpha bravo delta'. Break.
    # Returns "alpha bravo delta"


    # THIS IS THE SCENARIO!


    text_diff = "alpha bravo charlie_very_long delta"
    max_len_diff = 17 # Adjusted max_length to make 'delta' fit if charlie is skipped
    separator_diff = " "


    # With save_order=True:
    # "alpha " (6) + "bravo " (6) -> "alpha bravo " (12)
    # "charlie_very_long" (17) -> 12 + 17 = 29 > 17. Break due to save_order=True.
    assert smart_truncate(text_diff, max_len_diff, word_boundary=True, separator=separator_diff, save_order=True) == "alpha bravo"


    # With save_order=False:
    # "alpha " (6) + "bravo " (6) -> "alpha bravo " (12)
    # "charlie_very_long" (17) -> 12 + 17 = 29 > 17. Skip.
    # "delta" (5) -> 12 + 5 = 17. 17 == 17. Add "delta". Break.
    assert smart_truncate(text_diff, max_len_diff, word_boundary=True, separator=separator_diff, save_order=False) == "alpha bravo delta"

def parse_wheel_filename(
    filename: str,
    *,
    validate_order: bool = False,
) -> tuple[NormalizedName, Version, BuildTag, frozenset[Tag]]:
  
    if not filename.endswith(".whl"):
        raise InvalidWheelFilename(
            f"Invalid wheel filename (extension must be '.whl'): {filename!r}"
        )


    filename = filename[:-4]
    dashes = filename.count("-")
    if dashes not in (4, 5):
        raise InvalidWheelFilename(
            f"Invalid wheel filename (wrong number of parts): {filename!r}"
        )


    parts = filename.split("-", dashes - 2)
    name_part = parts[0]
    # See PEP 427 for the rules on escaping the project name.
    if "__" in name_part or re.match(r"^[\w\d._]*$", name_part, re.UNICODE) is None:
        raise InvalidWheelFilename(f"Invalid project name: {filename!r}")
    name = canonicalize_name(name_part)


    try:
        version = Version(parts[1])
    except InvalidVersion as e:
        raise InvalidWheelFilename(
            f"Invalid wheel filename (invalid version): {filename!r}"
        ) from e


    if dashes == 5:
        build_part = parts[2]
        build_match = _build_tag_regex.match(build_part)
        if build_match is None:
            raise InvalidWheelFilename(
                f"Invalid build number: {build_part} in {filename!r}"
            )
        build = cast("BuildTag", (int(build_match.group(1)), build_match.group(2)))
    else:
        build = ()
    tag_str = parts[-1]
    try:
        tags = parse_tag(tag_str, validate_order=validate_order)
    except UnsortedTagsError:
        raise InvalidWheelFilename(
            f"Invalid wheel filename (compressed tag set components must be in "
            f"sorted order per PEP 425): {filename!r}"
        ) from None
    return (name, version, build, tags)
