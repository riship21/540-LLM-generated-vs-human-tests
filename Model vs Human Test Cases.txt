def is_string_dtype(arr_or_dtype) -> bool:
    """
    Check whether the provided array or dtype is of the string dtype.


    Parameters
    ----------
    arr_or_dtype : array-like
        The array or dtype to check.


    Returns
    -------
    boolean
        Whether or not the array or dtype is of the string dtype.


    Examples
    --------
    >>> is_string_dtype(str)
    True
    >>> is_string_dtype(object)
    True
    >>> is_string_dtype(int)
    False
    >>>
    >>> is_string_dtype(np.array(['a', 'b']))
    True
    >>> is_string_dtype(pd.Series([1, 2]))
    False
    """
    # TODO: gh-15585: consider making the checks stricter.
    def condition(dtype) -> bool:
        return dtype.kind in ("O", "S", "U") and not is_excluded_dtype(dtype)


    def is_excluded_dtype(dtype) -> bool:
        """
        These have kind = "O" but aren't string dtypes so need to be explicitly excluded
        """
        is_excluded_checks = (is_period_dtype, is_interval_dtype)
        return any(is_excluded(dtype) for is_excluded in is_excluded_checks)


    return _is_dtype(arr_or_dtype, condition)

class _Numeric:
    """Class to easily allow comparing numbers"""

    def __init__(self, specifier):
        self.specifier = int(specifier)

    def __repr__(self):
        return repr(self.specifier)

    def __eq__(self, other):
        if isinstance(other, _Numeric):
            return self.specifier == other.specifier
        elif isinstance(other, int):
            return self.specifier == other

        return False

    def __ne__(self, other):
        return not self.__eq__(other)

    def __lt__(self, other):
        if isinstance(other, _Numeric):
            return self.specifier < other.specifier
        elif isinstance(other, int):
            return self.specifier < other
        elif isinstance(other, _Alpha):
            return True

        raise ValueError

class TestNumericComparisonOperators:
    """
    Regression tests for comparison operators in _Numeric.
    """

    def test_numeric_greater_than_itself_is_false(self):
        """
        Ensures that a _Numeric instance is not considered greater than itself.
        This directly targets the bug in _Numeric.__gt__ where it was incorrectly
        implemented as 'not self.__lt__(other)', causing '5 > 5' to return True.
        """
        num = _Numeric(5)
        # In the buggy version:
        # num.__lt__(num) -> False (since 5 < 5 is False)
        # num.__gt__(num) -> not False -> True
        # So, 'num > num' evaluates to True.
        # The assertion 'assert not (num > num)' will then fail because 'not True' is False.
        # In the fixed version, num.__gt__(num) returns False, so 'not False' is True, and the test passes.
        assert not (num > num)

    def test_numeric_greater_than_or_equal_to_itself_is_true(self):
        """
        Ensures that a _Numeric instance is considered greater than or equal to itself.
        This tests the __ge__ operator. While __ge__ was indirectly affected by the
        bug in __gt__, for self-comparison, it happened to yield the correct result
        in the buggy version due to __eq__ being correct.
        This test is included for completeness but is not the primary failure point.
        """
        num = _Numeric(5)
        assert num >= num

def check_required_arguments(argument_spec, module_parameters):
    """Check all paramaters in argument_spec and return a list of parameters
    that are required but not present in module_parameters
    """
    missing = []
    if argument_spec is None:
        return missing

    for (k, v) in argument_spec.items():
        required = v.get('required', False)
        if required and k not in module_parameters:
            missing.append(k)

    if missing:
        msg = "missing required arguments: %s" % ", ".join(missing)
        raise TypeError(msg)

    return missing

def is_in_obj(gpr, obj):
    if not hasattr(gpr, "name"):
        return False
    try:
        return gpr is obj[gpr.name]
    except (KeyError, IndexError):
        return False

def test_is_in_obj_identity_and_edge_cases():
    class MockGPR:
        def __init__(self, name_attr=None):
            if name_attr is not None:
                self.name = name_attr

    # Case 1: gpr is exactly obj[gpr.name] -> Should return True
    gpr_target = MockGPR("item_name")
    container = {"item_name": gpr_target, "other_item": MockGPR("other_name")}
    assert is_in_obj(gpr_target, container) is True

    # Case 2: gpr is NOT obj[gpr.name], even if they have the same 'name' and value -> Should return False
    gpr_different_instance = MockGPR("item_name")
    assert is_in_obj(gpr_different_instance, container) is False

    # Case 3: gpr has no 'name' attribute -> Should return False
    gpr_no_name = MockGPR()
    assert is_in_obj(gpr_no_name, container) is False

    # Case 4: gpr.name is not found in obj (KeyError/IndexError) -> Should return False
    gpr_missing_key = MockGPR("non_existent_key")
    assert is_in_obj(gpr_missing_key, container) is False

    # Case 5: obj is a list and gpr.name is a valid index, and gpr is the object at that index -> Should return True
    gpr_list_item = MockGPR(0)
    list_container = [gpr_list_item, MockGPR(1)]
    assert is_in_obj(gpr_list_item, list_container) is True

    # Case 6: gpr.name is an invalid index for a list (IndexError) -> Should return False
    gpr_invalid_index = MockGPR(999)
    assert is_in_obj(gpr_invalid_index, list_container) is False

    # Case 7: obj is not subscriptable (e.g., None, int) -> Should return False
    assert is_in_obj(gpr_target, None) is False
    assert is_in_obj(gpr_target, 123) is False
