def validate_collection_path(collection_path):
    if os.path.split(collection_path)[1] != 'ansible_collections':
        return os.path.join(collection_path, 'ansible_collections')
    return collection_path

   """

   # Example: '/tmp/my_collections/ansible_collections/'

   collection_path_with_trailing_slash = base_path + os.path.sep
   # In the buggy version of validate_collection_path:
   # 1. os.path.split('/tmp/test_data/ansible_collections/')[1] will be '' (empty string).

   # thereby exposing the bug in the buggy version.
   assert result == expected_buggy_output

def is_string_dtype(arr_or_dtype) -> bool:
    """
    Check whether the provided array or dtype is of the string dtype.


    Parameters
    ----------
    arr_or_dtype : array-like
        The array or dtype to check.


    Returns
    -------
    boolean
        Whether or not the array or dtype is of the string dtype.


    Examples
    --------
    >>> is_string_dtype(str)
    True
    >>> is_string_dtype(object)
    True
    >>> is_string_dtype(int)
    False
    >>>
    >>> is_string_dtype(np.array(['a', 'b']))
    True
    >>> is_string_dtype(pd.Series([1, 2]))
    False
    """
    # TODO: gh-15585: consider making the checks stricter.
    def condition(dtype) -> bool:
        return dtype.kind in ("O", "S", "U") and not is_excluded_dtype(dtype)


    def is_excluded_dtype(dtype) -> bool:
        """
        These have kind = "O" but aren't string dtypes so need to be explicitly excluded
        """
        is_excluded_checks = (is_period_dtype, is_interval_dtype)
        return any(is_excluded(dtype) for is_excluded in is_excluded_checks)


    return _is_dtype(arr_or_dtype, condition)

def parse_dfxp_time_expr(time_expr):
if not time_expr:
return


mobj = re.match(r'^(?P<time_offset>\d+(?:\.\d+)?)s?$', time_expr)
if mobj:
return float(mobj.group('time_offset'))

class TestUtils:
    def test_match_str_boolean_unary_filter(self):
        # The bug in the original version of `_match_one` causes unary operators
        # to treat boolean `False` values incorrectly.
        #
        # Buggy behavior:
        # - Unary `''` (presence check): `lambda v: v is not None`.
        #   For `v=False`, `False is not None` evaluates to `True`.
        #   This means `'my_bool'` would match `{'my_bool': False}`, which is incorrect.
        # - Unary `'!'` (absence check): `lambda v: v is None`.
        #   For `v=False`, `False is None` evaluates to `False`.
        #   This means `'!my_bool'` would NOT match `{'my_bool': False}`, which is incorrect.


        # This assertion should fail in the buggy version.
        # It expects `my_bool` (which is False) to NOT pass the presence filter.
        assert not match_str('my_bool', {'my_bool': False})


        # This assertion should fail in the buggy version.
        # It expects `!my_bool` (which is False) to pass the absence filter.
        assert match_str('!my_bool', {'my_bool': False})


        # These cases are for completeness and pass in both buggy and fixed versions.
        # Checking for presence of a True boolean field
        assert match_str('my_bool', {'my_bool': True})


        # Checking for absence of a True boolean field
        assert not match_str('!my_bool', {'my_bool': True})
